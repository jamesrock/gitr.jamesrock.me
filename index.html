git. simplified.

install on macOS

curl -O http://gitr.jamesrock.me/bin/gitr;chmod 755 gitr;sudo mv gitr /usr/local/bin

create new example branch from current branch

gitr branch example

delete example branch

gitr unbranch example

checkout master branch

gitr checkout master

merge example branch into current branch

gitr merge example

commit changes to current branch

gitr commit "some work"

discard changes to specific file since last commit

gitr discard js/app.js

discard changes to entire branch since last commit

gitr discard

revert specific file to commit-specific version

gitr revert js/app.js 43fba2...

revert entire branch to commit-specific version

gitr revert 43fba2...

push commits to remote

gitr push

pull commits from remote

gitr pull

list all commits (hit q to exit!)

gitr commits

list all branches

gitr branches

see git status

gitr status

update remotes

gitr update
